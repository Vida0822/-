let 가족관계 = '{
"할아버지" : {이름 : 홍기수, 나이: 80} ,
"할머니" : {이름: 홍기수, 나이:80} ,
"장남":"홍길동",
"차남": "홍길동",
"장녀":"홍길동"
}'; 

alert(typeof 가족관계); 
// json 표기법으로 만들어진 String (Object 아님!)
**string -> js 객체로 변환해줘야함 : JSON.parse(가족관계) ; 
(string안의 데이터를 떼올 수 없기때문에 객체로 변환해야 '쌍'을 가진 데이터임을 인식 )
const obj = JSON.parse(가족관계)
obj.장남 
obj.할아버지.이름
